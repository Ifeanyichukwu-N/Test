import express from "express"; import cors from "cors"; const app = express(); app.use(cors()); app.use(express.json()); /* ========================= SIMPLE MEMORY DATABASE ========================= */ let users = []; let videos = [ { id: 1, title: "Welcome to VIDSTREAM", url: "https://www.w3schools.com/html/mov_bbb.mp4", likes: 0, comments: [] } ]; let currentUser = null; /* ========================= AUTH ========================= */ app.post("/register", (req, res) => { users.push(req.body); res.json("User registered"); }); app.post("/login", (req, res) => { const user = users.find( u => u.email === req.body.email && u.password === req.body.password ); if (!user) return res.json("Invalid login"); currentUser = user; res.json({ message: "Logged in", user }); }); /* ========================= VIDEOS ========================= */ app.get("/videos", (req, res) => { res.json(videos); }); app.post("/videos", (req, res) => { videos.push({ id: videos.length + 1, title: req.body.title, url: req.body.url, likes: 0, comments: [] }); res.json("Video added"); }); app.put("/like/:id", (req, res) => { const video = videos.find(v => v.id == req.params.id); video.likes++; res.json(video); }); app.put("/comment/:id", (req, res) => { const video = videos.find(v => v.id == req.params.id); video.comments.push({ user: currentUser?.email || "guest", text: req.body.text }); res.json(video); }); /* ========================= FRONTEND (SINGLE PAGE) ========================= */ app.get("/", (req, res) => { res.send(` VIDSTREAM

🎬 VIDSTREAM

Register / Login

Upload Video

Videos

`); }); /* ========================= START SERVER ========================= */ app.listen(5000, () => { console.log("VIDSTREAM running on http://localhost:5000"); });
